    function startWatch() 
    {
    	console.log("Entering startWatch");
            
    	//Has Cordova initialized?
        	if (cvaReady) 
    	{
          	//Do we have a watch already running?
              	if (!watchID) 
    		{
                	//Update the screen 
    			//so the user knows the watch is starting
                	//Blank out any pevious skip alert warning
                	skipAlert.innerHTML = "";
                	//It might take a few seconds 
    			//for the first results to appear
                	li.innerHTML = 'Starting Watch';
                	//start the watch
                	watchID = navigator.geolocation.watchPosition(locationSuccess, locationError, locationOptions);
              	} 
    		else //watch running
    		{
                	console.log("Watch is already active");
                	navigator.notification.alert("A watch is already enabled.\nPlease don't tap the Start button.", null, alertTitle, alertBtn);
              	}
      	} 
    	else // cordova has not initialized
    	{
              console.log("Cordova is not ready");
              navigator.notification.alert("Please wait until Cordova has initialized.", null, alertTitle, alertBtn);
         	}
        	console.log("Leaving startWatch");
    }//end start watch
    
    function cancelWatch() 
    {
    	console.log("Entering cancelWatch");
            
    	//Do we have a watch active?
         	if (watchID) 
    	{
         		//Clear the watch
              	navigator.geolocation.clearWatch(watchID);
              	//Clear the watch ID (just because)
              	watchID = null;
              	//Clear out the last location results
              	li.innerHTML = '';
              	//Blank out the skip alert warning
              	skipAlert.innerHTML = "";
              	navigator.notification.alert("Watch cancelled", null, alertTitle, alertBtn);
       	} 
    	else 
    	{
              	console.log("Watch not active");
              	navigator.notification.alert("A watch is not active, nothing to cancel.\nPlease don't tap the Cancel button.", null, alertTitle, alertBtn);
        	}
         	console.log("Entering cancelWatch");
    }// end cancelWatch

    function gotMovement(a) 
    {
    	if(!lastX) {
    		lastX = a.x;
    		lastY = a.y;
    		lastZ = a.z;
    		return;
    	}
    	var deltaX, deltaY, deltaZ;
    	deltaX = Math.abs(a.x-lastX);
    	deltaY = Math.abs(a.y-lastY);
    	deltaZ = Math.abs(a.z-lastZ);
    	if(deltaX + deltaY + deltaZ > 3) {
    		moveCounter++;
    	} else {
    		moveCounter = Math.max(0, --moveCounter);
    	}
    	if(deltaX !=0 || deltaY != 0 || deltaZ != 0) console.log(deltaX,deltaY,deltaZ,moveCounter);
    	if(moveCounter > 1) { shuffleNumber(); moveCounter=0; }
    	lastX = a.x;
    	lastY = a.y;
    	lastZ = a.z;
    }

    }
    
    function getRandomInt(min, max) 
    {
      	return Math.floor(Math.random() * (max - min + 1)) + min;
    }

    function clickCounter() {
        if(typeof(Storage) !== "undefined") {
            if (localStorage.clickcount) {
                localStorage.clickcount = Number(localStorage.clickcount)+1;
            } else {
                localStorage.clickcount = 1;
            }
            document.getElementById("result").innerHTML = "Your favorite location is: " + localStorage.clickcount + ".";
        } else {
            document.getElementById("result").innerHTML = "Sorry, your browser does not support web storage...";
        }
    }

      <h2>FIT Campus Tour App</h2> This is an application that uses the lessons from CIS 5400. It uses the following API from Cordova: Accelerometer, Camera, Capture, Events, File Permissions, Geolocation, Media, and Splash Screen.
      <br>
      <br> It also uses the JQuery Mobile interface library to enhance the Graphical User Interface of the Application.
      <br>
      <br>
      <h3>The App:</h3>
      <br> There is an icon with a picture of the Botanical Gardens. I changed the default icon and resized my picture to the appropriate sizes.
      <br>
      <br> When the Application opens a Splash Screen of the Botanical Gardens is displayed. This uses the Cordova function to display a splash screen from example 16.1. To change the default splash screen, I added my own picture to the application directories
      and resized it to the different screen sizes.
      <br>
      <br> An event message is displayed for the Cordova &ldquo;onBodyLoad&rdquo; function from Cordova chapter 1. This function registers an event listener that calls the &ldquo;onDeviceReady&rdquo; function when the device is ready.
      <br>
      <br> When the device is ready, an event message is displayed to let you know the application is ready.
      <br>
      <br>
      <h3>Home:</h3>
      <br> The &ldquo;Home&rdquo; screen of the application displays a static google map of the FIT campus with markers at different locations.
      <br>
      <br> Home has:
      <br> • &ldquo;Start&rdquo; and &rdquo;Finish&rdquo; buttons to display your current location. This uses the Cordova Cookbook function from 11.2 to display the longitude and latitude. It watches your location.
      <br>
      <br> • Location buttons for each location display the Location name, longitude and Latitude.
      <br>
      <br> • A random location number is displayed based on a function that uses the Cordova Accelerometer.
      <br>
      <br> • Favorite Location Buttons: &ldquo;Show&rdquo;, &ldquo;+&rdquo; and &ldquo;- &rdquo;, which enable you to select your favorite number. This is saved in local storage and will be there the next time you use the app.
      <br>
      <br> • Header and Footer with links to home, back, and other pages.
      <br>
      <br>
      <h3>About: </h3>
      <br> Is a description of the App and Contact Information.
      <br>
      <br>
      <h3>Photo: </h3>
      <br> Enables the user to take a picture with the Camera API and display the most recent photo.
      <br>
      <br>
      <h3>Video: </h3>
      <br> Enables the user to make a video with the Capture API and displays the most recent video.
      <br>
      <br>
      <h3>Info: </h3>
